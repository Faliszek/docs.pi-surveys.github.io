/** @module App
 * 
 * 
 *  **/


 /**
  * Main file of node js backend application
  */



/**
  * Environmental variable - use development for test or NODE_ENV for deloying  */
const env = process.env.NODE_ENV || 'development';

/** database connection confif */
const config = require('./config/mongo')[env];

/** the most popular node js framework to easily create web apps */
const express = require('express')

/** Node.js body parsing middleware. */
const bodyParser = require("body-parser");

/** ORM for mongoDB database. */
const mongoose = require("mongoose");

/** environmental database url */
const envUrl = process.env[config.use_env_variable];

/** local database url */
const DATABASE_URL = `mongodb://${config.host}/${config.database}`;

/** final database url */
const MONGO_URL = envUrl ? envUrl : DATABASE_URL;

const graphqlHttp = require("express-graphql");
const graphqlSchema = require("./graphql/schema");
const graphqlResolver = require("./graphql/resolvers");

const auth = require("./middleware/auth");

const app = express();

app.use(bodyParser.json());

app.use((req, res, next) => {
  res.setHeader("Access-Control-Allow-Origin", "*");
  res.setHeader(
    "Access-Control-Allow-Methods",
    "OPTIONS, GET, POST, PUT, PATCH, DELETE"
  );
  res.setHeader("Access-Control-Allow-Headers", "Content-Type, Authorization");
  if (req.method === "OPTIONS") {
    return res.sendStatus(200);
  }
  next();
});

/** authentication middleware */
app.use(auth);


/** graphql middleware */
app.use(
  "/graphql",
  graphqlHttp({
    schema: graphqlSchema,
    rootValue: graphqlResolver,
    graphiql: true,
    formatError(err) {
      if (!err.originalError) {
        return err;
      }
      const data = err.originalError.data;
      const message = err.message || "An error occurred.";
      const code = err.originalError.code || 500;
      return { message: message, status: code, data: data };
    },
  })
);

/** errors middleware */
app.use((error, req, res, next) => {
  console.log(error, "error");
  const status = error.statusCode || 500;
  const message = error.message;
  const data = error.data;
  res.status(status).json({ message: message, data: data });
});
